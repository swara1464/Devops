#Create & Switch to New Branch
git checkout -b update-form

#Modify index.html
<label>Department: <input type="text" name="department" required></label><br>

#Stage & Commit Changes
git add index.html
git commit -m "Add Department field to registration form"

#Switch Back to Main
git checkout main

#Merge Branch into Main
git merge update-form

#Push Updated Main to GitHub
git push origin main

#Optional: Delete Branch (Local + Remote)
git branch -d update-form
git push origin --delete update-form
